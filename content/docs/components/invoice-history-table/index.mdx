```tsx
import { InvoiceHistoryTable } from '@/components/billingsdk/invoice-history-table';

<InvoiceHistoryTable
	title="Billing History"
	description="Your past invoices and payment records"
	invoices={[
		{
			id: '1',
			date: '2025-08-15',
			description: 'Pro Plan (Monthly)',
			amount: '$29.00',
			status: 'paid',
		},
		{
			id: '2',
			date: '2025-07-15',
			description: 'Pro Plan (Monthly)',
			amount: '$29.00',
			status: 'paid',
		},
	]}
	onDownload={(id) => console.log(`Download invoice ${id}`)}
/>;
```

## Props

| Prop        | Type                   | Description                                      |
| ----------- | ---------------------- | ------------------------------------------------ |
| className   | `string`               | Additional CSS classes to apply to the component |
| title       | `string`               | Title for the invoice history section            |
| description | `string`               | Description for the invoice history section      |
| invoices    | `Invoice[]`            | Array of invoice data                            |
| onDownload  | `(id: string) => void` | Function called when download button is clicked  |

### Invoice Object

| Property    | Type                                              | Description                       |
| ----------- | ------------------------------------------------- | --------------------------------- |
| id          | `string`                                          | Unique identifier for the invoice |
| date        | `string`                                          | Date of the invoice               |
| description | `string`                                          | Description of the invoice        |
| amount      | `string`                                          | Amount of the invoice             |
| status      | `"paid" \| "pending" \| "overdue" \| "cancelled"` | Status of the invoice             |
