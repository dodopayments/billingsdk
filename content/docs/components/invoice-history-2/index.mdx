## Usage

```tsx
import { InvoiceHistory2 } from '@/components/billingsdk/invoice-history-2';
```

```tsx
<InvoiceHistory2
	title="Billing History"
	description="Your past invoices and payment records"
	invoices={[
		{
			id: '1',
			date: '2025-08-15',
			description: 'Pro Plan (Monthly)',
			amount: '$29.00',
			status: 'paid',
			invoiceUrl: 'https://example.com/invoice/1', // Optional URL for direct viewing
		},
		{
			id: '2',
			date: '2025-07-15',
			// description is now optional
			amount: '$29.00',
			status: 'paid',
			// Action will use onDownload callback when invoiceUrl is not provided
		},
	]}
	actions="download,view"
	onDownload={(id) => console.log(`Download invoice ${id}`)}
	onView={(id) => console.log(`View invoice ${id}`)}
/>;
```

## Props

| Prop        | Type                                 | Description                                                                                       |
| ----------- | ------------------------------------ | ------------------------------------------------------------------------------------------------- |
| className   | `string`                             | Additional CSS classes to apply to the component                                                  |
| title       | `string`                             | Title for the invoice history section                                                             |
| description | `string`                             | Description for the invoice history section                                                       |
| invoices    | `Invoice[]`                          | Array of invoice data                                                                             |
| actions     | `('download' \| 'view')[] \| string` | Controls which action buttons are shown. Accepts array or comma string. Defaults to both actions. |
| onDownload  | `(id: string) => void`               | Called when Download is clicked                                                                   |
| onView      | `(id: string) => void`               | Called when View is clicked and no `invoiceUrl` is provided                                       |

### Invoice Object

| Property    | Type                                       | Description                                         |
| ----------- | ------------------------------------------ | --------------------------------------------------- |
| id          | `string`                                   | Unique identifier for the invoice                   |
| date        | `string`                                   | Date of the invoice                                 |
| description | `string` _optional_                        | Description of the invoice                          |
| amount      | `string`                                   | Amount of the invoice                               |
| status      | `"paid" \| "refunded" \| "open" \| "void"` | Status of the invoice                               |
| invoiceUrl  | `string` _optional_                        | URL to view the invoice directly (takes precedence) |

### Example

```tsx
<InvoiceHistory2
	title="Invoice History"
	description="All your billing records"
	invoices={[
		{
			id: 'inv-001',
			date: '2025-09-15',
			description: 'Enterprise Plan (Annual)',
			amount: '$1188.00',
			status: 'paid',
			invoiceUrl: 'https://example.com/invoice/inv-001',
		},
		{
			id: 'inv-002',
			date: '2025-08-15',
			description: 'Enterprise Plan (Annual)',
			amount: '$1188.00',
			status: 'refunded',
		},
		{
			id: 'inv-003',
			date: '2025-07-15',
			// description is optional
			amount: '$1188.00',
			status: 'open',
			invoiceUrl: 'https://example.com/invoice/inv-003',
		},
	]}
	onDownload={(invoiceId) => {
		// Handle invoice download
		downloadInvoice(invoiceId);
	}}
/>
```
