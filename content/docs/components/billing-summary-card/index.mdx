```tsx
import { BillingSummaryCard } from '@/components/billingsdk/billing-summary-card';

<BillingSummaryCard
	plan={{
		name: 'Pro',
		price: '$29',
		period: 'month',
		renewalDate: 'Oct 15, 2025',
		paymentMethod: 'Visa ending in 4242',
	}}
	onUpgrade={() => console.log('Upgrade clicked')}
/>;
```

## Props

| Prop      | Type         | Description                                        |
| --------- | ------------ | -------------------------------------------------- |
| className | `string`     | Additional CSS classes to apply to the card        |
| plan      | `Plan`       | Object containing plan information                 |
| onUpgrade | `() => void` | Function called when the upgrade button is clicked |

### Plan Object

| Property      | Type     | Description                            |
| ------------- | -------- | -------------------------------------- |
| name          | `string` | Name of the current plan               |
| price         | `string` | Price of the plan                      |
| period        | `string` | Billing period (e.g., "month", "year") |
| renewalDate   | `string` | Date when the subscription renews      |
| paymentMethod | `string` | Description of the payment method      |

```tsx
<BillingSummaryCard
	plan={{
		name: 'Enterprise',

		period: 'month',

		paymentMethod: 'Mastercard ending in 5555',
	}}
	onUpgrade={() => {
		// Handle upgrade logic
	}}
/>
```
